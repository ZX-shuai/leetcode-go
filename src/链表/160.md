```Golang
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func getIntersectionNode(headA, headB *ListNode) *ListNode {
    var(
        lenA int
        lenB int
        count int
        fast *ListNode
        slow *ListNode
    )
    
    for headA != nil {//这边最好用两个指针代替头指针进行遍历，不然后面用到头指针，发现头指针没有指向第一个元素
        headA = headA.Next
        lenA++
    }
    for headB != nil {
        headB = headB.Next
        lenB++
    }

    if lenA > lenB{
        count = lenA-lenB
        fast,slow = headB,headA
    }else{
        count = lenB-lenA
        fast,slow = headA,headB
    }

    for i:=0;i<=count;i++{//应为i<count
        slow = slow.Next
    }
    for fast != slow{
        fast = fast.Next
        slow = slow.Next
    }
    return fast
}
```
修改后的代码如下：
```Golang
func getIntersectionNode(headA, headB *ListNode) *ListNode {
    var(
        lenA int
        lenB int
        count int
        fast *ListNode
        slow *ListNode
    )
    curA := headA
    curB := headB
    for curA != nil {
        curA = curA.Next
        lenA++
    }
    for curB != nil {
        curB = curB.Next
        lenB++
    }

    if lenA > lenB{
        count = lenA-lenB
        fast,slow = headB,headA
    }else{
        count = lenB-lenA
        fast,slow = headA,headB
    }

    for i:=0;i<count;i++{//应为i<count
        slow = slow.Next
    }
    for fast != slow{
        fast = fast.Next
        slow = slow.Next
    }
    return fast
}
```
