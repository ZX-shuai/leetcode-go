/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func getIntersectionNode(headA, headB *ListNode) *ListNode {
    var(
        lenA int
        lenB int
        count int
        fast *ListNode
        slow *ListNode
    )
    
    for headA != nil {
        headA = headA.Next
        lenA++
    }
    for headB != nil {
        headB = headB.Next
        lenB++
    }

    if lenA > lenB{
        count = lenA-lenB
        fast,slow = headB,headA
    }else{
        count = lenB-lenA
        fast,slow = headA,headB
    }

    for i:=0;i<=count;i++{
        slow = slow.Next
    }
    for fast != slow{
        fast = fast.Next
        slow = slow.Next
    }
    return fast
}
